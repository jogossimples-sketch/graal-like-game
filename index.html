class Jogo: def __init__(self, nome): self.nome = nome self.personagens = [] self.edificios = [] def adicionar_personagem(self, personagem): self.personagens.append(personagem) def adicionar_edificio(self, edificio): self.edificios.append(edificio) class Personagem: def __init__(self, nome, tipo): self.nome = nome self.tipo = tipo self.inventario = [] def pegar_item(self, item): self.inventario.append(item) # PARTE NOVADA ADICIONADA - NENHUMA ALTERACAO NO CODIGO ACIMA! # Item: Picareta class Picareta: def __init__(self, tipo="padrão"): self.nome = "Picareta" self.tipo = tipo self.forca_mineracao = 5 if tipo == "padrão" else 10 if tipo == "ferro" else 15 def __str__(self): return f"Picareta de {self.tipo} (Força: {self.forca_mineracao})" # Personagem: Mineiro class Mineiro(Personagem): def __init__(self, nome, nivel=1): super().__init__(nome, tipo="Mineiro") self.nivel = nivel self.picaretas_disponiveis = [Picareta(), Picareta("ferro"), Picareta("diamante")] self.picareta_atual = self.picaretas_disponiveis[0] def trocar_picareta(self, tipo_picareta): for picareta in self.picaretas_disponiveis: if picareta.tipo == tipo_picareta: self.picareta_atual = picareta print(f"{self.nome} trocou para picareta de {tipo_picareta}!") return print(f"Picareta de {tipo_picareta} não encontrada!") def minerar(self): print(f"{self.nome} está minerando com {self.picareta_atual}...") return self.picareta_atual.forca_mineracao * self.nivel # Edificio: Casa de Mineração class CasaMineracao: def __init__(self, nome="Casa de Mineração", localizacao="Montanha"): self.nome = nome self.localizacao = localizacao self.mineiros_atuantes = [] self.recursos_minerados = 0 def contratar_mineiro(self, mineiro): if mineiro.tipo == "Mineiro": self.mineiros_atuantes.append(mineiro) print(f"{mineiro.nome} foi contratado para trabalhar na {self.nome}!") else: print(f"{mineiro.nome} não é um mineiro, não pode trabalhar aqui!") def realizar_mineracao_coletiva(self): print(f"\n--- Mineração na {self.nome} ({self.localizacao}) ---") for mineiro in self.mineiros_atuantes: recursos = mineiro.minerar() self.recursos_minerados += recursos print(f"Total de recursos minerados: {self.recursos_minerados}!\n") #
